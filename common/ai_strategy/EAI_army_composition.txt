############################################################################################################################################

#	Army composition scripts

#	-an army has a base of infantry, marines or mountaineers
#	-this base is modified by army focuses chosen by the AI on startup or by army focus presets in scripted triggers

#	-'attachments' of infantry, marines, mountaineers, AT infantry are added to the the base infantry composition based on need or army focuses

#	-AI chooses between a heavy armor and a medium armor focused armored force or AT focus
#	-a smaller 'attachment' of motorized, heavy armor or medium armor is added to the armored force

#	unit tokens:

#	infantry = infantry
#	marines = marines
#	mountaineers = mountaineers
#	paratroopers = paratroopers
#	cavalry = unused/light armor
#	armor = medium armor
#	mechanized = heavy armor
#	motorized = AT infantry

#	fighter = fighter
#	cas	= cas
#	tactical_bomber = tactical bomber
#	naval_bomber = naval bomber
#	interceptor = cv fighter + cv naval bomber
#	strategic_bomber = strategic_bomber
#	air_transport = disabled (transport, h. fighter, everything else)

#	capital_ship = battleship,  heavy battleship
#	screen_ship = destroyer, light cruiser
#	carrer = carrier
#	submarine = disabled (subs, BC, CA)

			}
		}

		OR = {
