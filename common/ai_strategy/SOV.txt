NAP_sov_per = {
	enable = {
		is_historical_focus_on = yes
		OR = { tag = SOV tag = PER }
		SOV = { is_ai = yes } PER = { is_ai = yes }
		NOT = { has_non_aggression_pact_with = PER has_non_aggression_pact_with = SOV }
		NOT = { is_in_faction_with = GER }
	
		OR = {
			GER = { has_war_with = SOV }
			GER = { has_wargoal_against = SOV }
			GER = { is_justifying_wargoal_against = SOV }
			GER = { focus_progress = { focus = GER_ussr_war_goal progress > 0.01 } }
		}
	}
	abort = {
		OR = {
			has_non_aggression_pact_with = PER
			has_non_aggression_pact_with = SOV
			is_in_faction_with = GER
		}
	}
	ai_strategy = { type = befriend id = "SOV" value = 1000 }
	ai_strategy = { type = befriend id = "PER" value = 1000 }
}
