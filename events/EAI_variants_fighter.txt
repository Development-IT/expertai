#variant
country_event = {
	id = EAI_VF.0
	hidden = yes
	
	mean_time_to_happen = { days = 2 }
	
	trigger = {
		is_ai = yes has_war = yes date > 1939.1.1
		has_tech = fighter1 NOT = { has_tech = fighter2 }
		NOT = { has_country_flag = f1_chain_triggered }
		enemies_strength_ratio > 0.5
	}
	
	immediate = {
		set_country_flag = f1_chain_triggered
		country_event = EAI_VF.1
	}
	
	option = { name = "OK" }
}

country_event = {
	id = EAI_VF.1
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_tech = fighter1 NOT = { has_tech = fighter2 } }
			create_equipment_variant = { 
				name = "FTR I A" 
				type = fighter_equipment_1
				upgrades = { 
					plane_engine_upgrade = 1
					plane_gun_upgrade = 0
					plane_reliability_upgrade = 1
				}
			}
			country_event = { id = EAI_VF.2 days = 20 random = 10 }
			air_experience = -55
		}
	}
	
	option = { name = "OK" }
}
