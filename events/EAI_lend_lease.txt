country_event = { #select a LL target
	id = EAI_LL.0
	hidden = yes
	mean_time_to_happen = { days = 5 }
	trigger = {
		has_global_flag = EAI_LL_enabled
	
		is_ai = yes
		OR = { #LL requirements for sender
			AND = { has_government = democratic threat > 0.5 }
			AND = { has_government = neutrality threat > 0.4 }
			NOT = { has_government = democratic has_government = neutrality }
		}
		NOT = { has_idea = neutrality_idea } #neutrals don't send LL (sweden, switzerland)
		OR = { is_major = yes is_in_faction = yes has_war = yes } #limit random minors from sending equipment all over the place
		
		any_other_country = {
			OR = { 
				original_tag = ENG original_tag = FRA original_tag = USA original_tag = GER original_tag = ITA original_tag = JAP original_tag = SOV 
				original_tag = BUL original_tag = ROM original_tag = HUN original_tag = CAN original_tag = AST original_tag = SAF original_tag = RAJ original_tag = NZL original_tag = POL original_tag = YUG original_tag = GRE original_tag = TUR 
				original_tag = CHI original_tag = MAN original_tag = MEN original_tag = SIA original_tag = MAL original_tag = INS original_tag = FIN original_tag = SPR original_tag = D01 
			}
			eligible_for_LL = yes
		}
	}
	immediate = {
		if = { #prioritise subjects
			limit = {
				any_other_country = {
					OR = { 
						original_tag = ENG original_tag = FRA original_tag = USA original_tag = GER original_tag = ITA original_tag = JAP original_tag = SOV 
						original_tag = BUL original_tag = ROM original_tag = HUN original_tag = CAN original_tag = AST original_tag = SAF original_tag = RAJ original_tag = NZL original_tag = POL original_tag = YUG original_tag = GRE original_tag = TUR 
						original_tag = CHI original_tag = MAN original_tag = MEN original_tag = SIA original_tag = MAL original_tag = INS original_tag = FIN original_tag = SPR original_tag = D01 
					}
					is_subject_of = ROOT
					eligible_for_LL = yes
				}
			}
			random_other_country = {
				limit = {
					OR = { 
						original_tag = ENG original_tag = FRA original_tag = USA original_tag = GER original_tag = ITA original_tag = JAP original_tag = SOV 
						original_tag = BUL original_tag = ROM original_tag = HUN original_tag = CAN original_tag = AST original_tag = SAF original_tag = RAJ original_tag = NZL original_tag = POL original_tag = YUG original_tag = GRE original_tag = TUR 
						original_tag = CHI original_tag = MAN original_tag = MEN original_tag = SIA original_tag = MAL original_tag = INS original_tag = FIN original_tag = SPR original_tag = D01 
					}
					is_subject_of = ROOT
					eligible_for_LL = yes
				}
				country_event = EAI_LL.1
			}
	
			else = {
				if = { #prioritise SOV
					limit = {
						any_other_country = {
							OR = { original_tag = SOV }
							eligible_for_LL = yes
						}
					}
					random_other_country = {
						limit = {
							OR = { original_tag = SOV }
							eligible_for_LL = yes
						}
						country_event = EAI_LL.1
					}
				
					else = {
						if = { #prioritise majors
							limit = {
								any_other_country = {
									OR = { original_tag = ENG original_tag = FRA original_tag = USA original_tag = GER original_tag = ITA original_tag = JAP }
									eligible_for_LL = yes
								}
							}
							random_other_country = {
								limit = {
									OR = { original_tag = ENG original_tag = FRA original_tag = USA original_tag = GER original_tag = ITA original_tag = JAP }
									eligible_for_LL = yes
								}
								country_event = EAI_LL.1
							}
							
							else = {
								random_other_country = {
									limit = {
										OR = {
											original_tag = BUL original_tag = ROM original_tag = HUN original_tag = CAN original_tag = AST original_tag = SAF original_tag = RAJ original_tag = NZL original_tag = POL original_tag = YUG original_tag = GRE original_tag = TUR 
											original_tag = CHI original_tag = MAN original_tag = MEN original_tag = SIA original_tag = MAL original_tag = INS original_tag = FIN original_tag = SPR original_tag = D01 
										}
										eligible_for_LL = yes
									}
									country_event = EAI_LL.1
								}
							}
						}
					}
				}
			}
		}
	}
	option = { }
}
