add_namespace = EAI_VCAS

#variant
country_event = {
	id = EAI_VCAS.0
	hidden = yes
	
	mean_time_to_happen = { days = 2 }
	
	trigger = {
		NOT = { tag = ENG tag = FRA tag = USA tag = ITA tag = JAP }
		is_ai = yes
		has_tech = CAS1
		NOT = { has_tech = CAS2 }
		has_war = yes
		NOT = { has_country_flag = cas1_chain_triggered }
		date > 1939.1.1
		enemies_strength_ratio > 0.5
	}
	
	immediate = {
		set_country_flag = cas1_chain_triggered
		country_event = EAI_VCAS.1
	}
	
	option = { name = "OK" }
}

country_event = {
	id = EAI_VCAS.1
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_tech = CAS1 NOT = { has_tech = CAS2 } }
			create_equipment_variant = { 
				name = "CAS I A" 
				type = CAS_equipment_1
				upgrades = {
					plane_engine_upgrade = 1
					plane_cas_upgrade = 1
					plane_reliability_upgrade = 0
				}
			}
			country_event = { id = EAI_VCAS.2 days = 20 random = 10 }
			air_experience = -55
		}
	}
	
	option = { name = "OK" }
}
