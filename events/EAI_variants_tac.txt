add_namespace = EAI_VTAC

#variant
country_event = {
	id = EAI_VTAC.0
	hidden = yes
	
	mean_time_to_happen = { days = 2 }
	
	trigger = {
		OR = { tag = ENG tag = FRA tag = USA tag = ITA tag = JAP }
		is_ai = yes
		has_tech = tactical_bomber1
		NOT = { has_tech = tactical_bomber2 }
		has_war = yes
		NOT = { has_country_flag = tac1_chain_triggered }
		date > 1939.1.1
		has_country_flag = f1_complete
		OR = {
			NOT = { has_country_flag = f2_chain_triggered }
			AND = { has_country_flag = f2_chain_triggered has_country_flag = f2_complete }
		}
		OR = {
			NOT = { has_country_flag = f3_chain_triggered }
			AND = { has_country_flag = f3_chain_triggered has_country_flag = f3_complete }
		}
		enemies_strength_ratio > 0.5
	}
	
	immediate = {
		set_country_flag = tac1_chain_triggered
		country_event = EAI_VTAC.1
	}
	
	option = { name = "OK" }
}

country_event = {
	id = EAI_VTAC.1
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_tech = tactical_bomber1 NOT = { has_tech = tactical_bomber2 } }
			create_equipment_variant = { 
				name = "TAC I A"
				type = tac_bomber_equipment_1
				upgrades = {
					plane_engine_upgrade = 1
					plane_tac_bomb_upgrade = 0
					plane_reliability_upgrade = 1
				}
			}
			country_event = { id = EAI_VTAC.2 days = 20 random = 10 }
			air_experience = -55
		}
	}
	
	option = { name = "OK" }
}
