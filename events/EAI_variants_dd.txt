add_namespace = EAI_VCL

#variant
country_event = {
	id = EAI_VCL.0
	hidden = yes
	
	mean_time_to_happen = { days = 2 }
	
	trigger = {
		is_ai = yes
		#NOT = { tag = GER }
		has_tech = basic_destroyer
		NOT = { has_tech = improved_destroyer }
		has_war = yes
		NOT = { has_country_flag = cl1_chain_triggered }
		date > 1939.1.1
		enemies_strength_ratio > 0.5
	}
	
	immediate = {
		set_country_flag = cl1_chain_triggered
		country_event = EAI_VCL.1
	}
	
	option = { name = "OK" }
}

country_event = {
	id = EAI_VCL.1
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_tech = basic_destroyer NOT = { has_tech = improved_destroyer } }
			create_equipment_variant = { 
				name = "Destroyer II A"
				type = destroyer_2
				upgrades = { 
					destroyer_engine_upgrade = 1
				}
			}
			country_event = { id = EAI_VCL.2 days = 30 }
			navy_experience = -20
		}
	}
	
	option = { name = "OK" }
}
