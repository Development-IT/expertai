country_event = {
	id = EAI_VSS.1
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_tech = basic_submarine NOT = { has_tech = improved_submarine } }
			create_equipment_variant = {
				name = "Submarine II A" 
				type = submarine_2
				upgrades = {
					sub_engine_upgrade = 1
				}
			}
			country_event = { id = EAI_VSS.2 days = 15 random = 10 }
			navy_experience = -25
		}
	}
	
	option = { name = "OK" }
}

		is_ai = yes

		has_war = yes

	}
	
	immediate = {
		set_country_flag = ss2_chain_triggered
		country_event = EAI_VSS.7
